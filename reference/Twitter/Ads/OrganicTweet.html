def stats(account, ids, metrics, start_time = nil, end_time = nil, granularity = :hour)
  resource = STATS_PATH % { account: account.id }
  end_time ||= Time.at((Time.now.to_f / 3600).round * 3600)
  start_time ||= (end_time - 86_400)
  params = {
    tweet_ids: ids.join(','),
    metrics: metrics.join(','),
    start_time: start_time.iso8601,
    end_time: end_time.iso8601,
    granularity: granularity.to_s.upcase
  }
  response = Request.new(account.client, :get, resource, params: params).perform

  if response.error?
    error = response.body[:errors].first
    fail StandardError.new("code: #{response.code} " \
                            "error: #{error[:code]} " \
                            "message: #{error[:message]}")
  end

  response.body[:data].each do |obj|
    obj.select! { |k, _v| metrics.include?(k.to_s) }
  end
end
